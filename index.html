বাংলা কোড রানার

বাংলা কোড লিখুন এবং রান করুন। সহজ ও ইন্টার‌্যাকটিভ সিস্টেমে আউটপুট দেখুন।


// Input
ধরি নাম = ইনপুট("তোমার নাম কি?");

// Condition
যদি (নাম == "আনিসুর") {
  দেখাও("হ্যালো আনিসুর!");
} নাহলে যদি (নাম == "লিখন") {
  দেখাও("হ্যালো লিখন!");
} নাহলে {
  দেখাও("ঘোড়ার ডিম! তোমার নাম কি?");
}

  নাও("%প", &নাম্বার);
  দেখাও("%প\n", নাম্বার);

}

//বাংলা ➝ সি কোড 
#include <stdio.h>
int main() {

}
